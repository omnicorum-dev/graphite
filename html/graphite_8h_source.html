1145 inline bool NormalizedRectangle::normalizeRectangle(const int32 x, const int32 y, const int32 w, const int32 h, const int32 canvas_width, const int32 canvas_height) {
1146 if (w == 0 || h == 0)
1147 return false;
1148
1149 // ---- STEP 1: compute ORIGINAL bounds (NO CLAMPING) ----
1150 ox1 = x; oy1 = y; ox2 = x + w; oy2 = y + h;
1151
1152 if (ox1 > ox2) std::swap(ox1, ox2);
1153 if (oy1 > oy2) std::swap(oy1, oy2);
1154
1155 // ---- STEP 2: cull using ORIGINAL bounds ----
1156 if (ox1 >= canvas_width || ox2 < 0 ||
1157 oy1 >= canvas_height || oy2 < 0)
1158 return false;
1159
1160 // ---- STEP 3: compute CLIPPED bounds ----
1161 x1 = glm::max(0, glm::min(ox1, canvas_width - 1));
1162 y1 = glm::max(0, glm::min(oy1, canvas_height - 1));
1163 x2 = glm::max(0, glm::min(ox2, canvas_width - 1));
1164 y2 = glm::max(0, glm::min(oy2, canvas_height - 1));
1165
1166 // ---- STEP 4: safety check (important) ----
1167 if (x1 > x2 || y1 > y2)
1168 return false;
1169
1170 return true;
1171 }
